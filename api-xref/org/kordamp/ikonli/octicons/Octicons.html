429     public static Octicons findByDescription(String description) {
430         for (Octicons font : values()) {
431             if (font.getDescription().equals(description)) {
432                 return font;
433             }
434         }
435         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
436     }
